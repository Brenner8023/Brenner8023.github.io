<body>
  <div id="app">
    <input type="button" value="浪起来" @click="start" />
    <input type="button" value="别浪" @click="stop" />
    <p v-cloak>{{msg}}</p>
  </div>
  <script src="https://cdn.staticfile.org/vue/2.6.7/vue.min.js"></script>
  <script>
    var vm = new Vue({
      el: "#app",
      data: {
        msg: "猥琐发育, 别浪!",
        intervalid: null
      },
      methods: {
        start(){
          if(this.intervalid != null)return 0;
          this.intervalid = setInterval(() => {
            var start = this.msg.substring(0,1);
            var end = this.msg.substring(1);
            this.msg = end + start;
          }, 200)
        },
        stop(){
          clearInterval(this.intervalid);
          this.intervalid = null;
        }
      }
    });
  </script>
</body>
