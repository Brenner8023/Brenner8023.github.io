    # echarts.js笔记

    参考来源: 官网: https://echarts.baidu.com/

    # 基础

    在绘图前我们需要为ECharts准备一个具备宽高的DOM容器

    柱状图

    <div id="echarts"></div>
<script>
  // 初始化echarts实例
	let myChart = echarts.init(document.getElementById('echarts'), 'dark'); // 设置颜色主题为dark
  
  // 指定图表的配置项和数据
  let options = {
  	title: {
    	text: 'ECharts入门示例'
    },
    legend: {
    	data: ['销量']
    },
    xAxis: {
    	data: ["衬衫","羊毛衫","雪纺衫","裤子","高跟鞋","袜子"]
    },
    yAxis: {},
    series: [{
    	name: '销量',
      type: 'bar',
      data: [5, 20, 36, 10, 10, 20],
      lebel: {
      	normal: {
					show: true,
					position: 'top',
					formatter: '{c} ℃'
      }
    }]
 	};
  
  myChart.setOption(options);
</script>

    # 个性化图表样式

    ECharts提供了丰富的自定义配置选项, 并且能够从全局, 系列, 数据三个层级去设置数据图形的样式.

    饼图

    let options = {
  series: [
    {
    	name: "访问来源",
      type: 'pie',
      roseType: 'angle', // 设置成南丁格尔图
      radius: '55%',
      backgroundColor: 'red',
      data: [
        {value: 235, name: '视频广告'},
        {value: 274, name: '联盟广告'},
        {value: 310, name: '邮件营销'},
        {value: 335, name: '直接访问'},
        {value: 400, name: '搜索引擎'}
      ]
    }
  ]
}

myChart.setOption(options)

    调色盘
    给定一组颜色, 图形会自动从其中选择颜色. 可以设置全局的调色盘或者某图表的调色盘

    option = {
    // 全局调色盘。
    color: ['#bda29a','#6e7074', '#546570', '#c4ccd3'],

    series: [{
        type: 'bar',
        // 此系列自己的调色盘。
        color: ['#73b9bc','#7289ab', '#91ca8c','#f49f42'],
    }, {
        type: 'pie',
        // 此系列自己的调色盘。
        color: ['#37A2DA', '#32C5E9', '#67E0E3',  '#8378EA', '#96BFFF'],
    }]
}

    # 异步数据加载和更新

    如果数据加载时间较长, 可以使用一个loading的动画来提示用户数据正在加载

    myChart.showLoading();
myChart.hideLoading(); // 数据加载完成后调用此方法隐藏加载动画
